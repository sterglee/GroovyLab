liniger1vs

Solves an initial value problem, given as an autonomous system of first order ordinary differential equations by means of an implicit, first-order accurate, exponentially fitted one-step method.

Example: Can be run with F9 within the gLab Editor (Ctrl-E to open the editor)

  public void derivative(int m, double a[], double sigma[])
  {
  	double a1,a2;

  	a1=a[1];
  	a2=a[2];
  	a[1]=(a1+0.99)*(a2-1.0)+0.99;
	  a[2]=1000.0*((1.0+a1)*(1.0-a2)-1.0);
  }


  public void jacobian(int m, double j[][], double y[],
                       double sigma[])
  {
  	j[1][1]=y[2]-1.0;
  	j[1][2]=0.99+y[1];
	  j[2][1]=1000.0*(1.0-y[2]);
  	j[2][2] = -1000.0*(1.0+y[1]);
	  sigma[0]=Math.abs(j[2][2]+j[1][1]-
             Math.sqrt((j[2][2]-j[1][1])*
             (j[2][2]-j[1][1])+4.0*j[2][1]*j[1][2]))/2.0;
  }
